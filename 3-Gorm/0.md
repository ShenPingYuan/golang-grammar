## 1. 安装与连接

```go
// 安装
// go get -u gorm.io/gorm
// go get -u gorm.io/driver/mysql
// go get -u gorm.io/driver/postgres
// go get -u gorm.io/driver/sqlite

package main

import (
    "time"
    "gorm.io/driver/mysql"
    "gorm.io/driver/postgres"
    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
)

func main() {
    // --- MySQL ---
    dsn := "user:pass@tcp(127.0.0.1:3306)/dbname?charset=utf8mb4&parseTime=True&loc=Local"
    db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})

    // --- PostgreSQL ---
    dsn2 := "host=localhost user=postgres password=secret dbname=mydb port=5432 sslmode=disable"
    db, err = gorm.Open(postgres.Open(dsn2), &gorm.Config{})

    // --- SQLite ---
    db, err = gorm.Open(sqlite.Open("test.db"), &gorm.Config{})

    if err != nil {
        panic("failed to connect database")
    }

    // 获取底层 *sql.DB，配置连接池
    sqlDB, _ := db.DB()
    sqlDB.SetMaxIdleConns(10)           // 最大空闲连接数
    sqlDB.SetMaxOpenConns(100)          // 最大打开连接数
    sqlDB.SetConnMaxLifetime(time.Hour) // 连接最大生命周期
}
```

## 2. 模型定义

```go
import "gorm.io/gorm"

// gorm.Model 内置字段：ID, CreatedAt, UpdatedAt, DeletedAt（软删除）
// type Model struct {
//     ID        uint           `gorm:"primarykey"`
//     CreatedAt time.Time
//     UpdatedAt time.Time
//     DeletedAt gorm.DeletedAt `gorm:"index"`
// }

// 基本模型
type User struct {
    gorm.Model                          // 内嵌 ID, CreatedAt, UpdatedAt, DeletedAt
    Name     string `gorm:"type:varchar(100);not null;index"`  // 带索引
    Email    string `gorm:"uniqueIndex;size:128"`              // 唯一索引
    Age      int    `gorm:"default:18"`                        // 默认值
    Active   bool   `gorm:"default:true"`
    Profile  Profile                    // Has One
    Orders   []Order                    // Has Many
}

// 自定义表名
func (User) TableName() string {
    return "t_users"
}

// 常用 struct tag
type Product struct {
    ID        uint    `gorm:"primaryKey;autoIncrement"`
    Code      string  `gorm:"type:varchar(50);uniqueIndex;not null;comment:产品编码"`
    Name      string  `gorm:"size:200;not null"`
    Price     float64 `gorm:"type:decimal(10,2);default:0"`
    Stock     int     `gorm:"check:stock >= 0"`                // CHECK 约束
    Category  string  `gorm:"index:idx_cat_price"`             // 复合索引（名称相同的归为一组）
    SalePrice float64 `gorm:"index:idx_cat_price"`
    Remark    string  `gorm:"type:text"`
    IgnoreMe  string  `gorm:"-"`                               // 忽略此字段
    ReadOnly  string  `gorm:"->"`                               // 只读
    WriteOnly string  `gorm:"<-"`                               // 只写
    CreateOnly string `gorm:"<-:create"`                        // 仅创建时可写
    UpdateOnly string `gorm:"<-:update"`                        // 仅更新时可写
}

// JSON 标签配合使用
type APIUser struct {
    gorm.Model
    Name  string `gorm:"size:100" json:"name"`
    Email string `gorm:"size:128" json:"email,omitempty"`
}

// 复合主键
type UserLanguage struct {
    UserID     uint   `gorm:"primaryKey"`
    LanguageID uint   `gorm:"primaryKey"`
    Skill      string
}

// 自定义字段类型映射
type Article struct {
    ID      uint
    Title   string
    Content string `gorm:"type:longtext"`          // MySQL longtext
    Tags    string `gorm:"type:json"`               // MySQL JSON 类型
}
```

## 3. 自动迁移

```go
// 自动创建表、补充缺失的列和索引
// 注意：不会删除列，不会降级类型
db.AutoMigrate(&User{}, &Product{}, &Order{})

// 检查表是否存在
db.Migrator().HasTable(&User{})
db.Migrator().HasTable("users")

// 手动建表/删表
db.Migrator().CreateTable(&User{})
db.Migrator().DropTable(&User{})

// 列操作
db.Migrator().AddColumn(&User{}, "Age")
db.Migrator().DropColumn(&User{}, "Age")
db.Migrator().AlterColumn(&User{}, "Name")
db.Migrator().HasColumn(&User{}, "Name")

// 索引操作
db.Migrator().CreateIndex(&User{}, "Name")
db.Migrator().DropIndex(&User{}, "idx_name")
db.Migrator().HasIndex(&User{}, "idx_name")

// 重命名
db.Migrator().RenameTable("users", "t_users")
db.Migrator().RenameColumn(&User{}, "Name", "FullName")
```

## 4. 创建（Create）

```go
// 创建单条记录
user := User{Name: "Alice", Age: 25, Email: "alice@go.dev"}
result := db.Create(&user)
// user.ID              — 回填主键
// result.Error         — 错误信息
// result.RowsAffected  — 插入行数

// 指定字段创建（只插入 Name 和 Age）
db.Select("Name", "Age").Create(&user)

// 忽略字段创建（除 Age 外都插入）
db.Omit("Age").Create(&user)

// 批量创建
users := []User{
    {Name: "Bob", Age: 20},
    {Name: "Carol", Age: 22},
    {Name: "Dave", Age: 30},
}
db.Create(&users)

// 分批插入（每批 100 条，避免超大 SQL）
db.CreateInBatches(&users, 100)

// 用 Map 创建（不触发 Hook、不回填、不处理关联）
db.Model(&User{}).Create(map[string]interface{}{
    "Name": "Eve", "Age": 28,
})

// 批量 Map 创建
db.Model(&User{}).Create([]map[string]interface{}{
    {"Name": "Frank", "Age": 30},
    {"Name": "Grace", "Age": 26},
})

// Upsert：冲突时更新（MySQL）
db.Clauses(clause.OnConflict{
    Columns:   []clause.Column{{Name: "email"}},        // 冲突判断列
    DoUpdates: clause.AssignmentColumns([]string{"name", "age"}), // 冲突时更新哪些列
}).Create(&users)

// Upsert：冲突时什么都不做
db.Clauses(clause.OnConflict{DoNothing: true}).Create(&user)
```

## 5. 查询（Query）

```go
var user User
var users []User

// --- 基础查询 ---

// 主键查询
db.First(&user, 1)                  // SELECT * FROM users WHERE id = 1 ORDER BY id LIMIT 1
db.First(&user, "id = ?", 10)

// 第一条（无排序保证）
db.Take(&user)

// 最后一条
db.Last(&user)

// 查询全部
db.Find(&users)

// --- Where 条件 ---

// 字符串条件
db.Where("name = ?", "Alice").First(&user)
db.Where("name <> ?", "Alice").Find(&users)
db.Where("name IN ?", []string{"Alice", "Bob"}).Find(&users)
db.Where("name LIKE ?", "%ali%").Find(&users)
db.Where("age BETWEEN ? AND ?", 20, 30).Find(&users)
db.Where("created_at > ?", time.Now().AddDate(0, 0, -7)).Find(&users)

// Struct 条件（零值字段会被忽略！）
db.Where(&User{Name: "Alice", Age: 0}).Find(&users)
// 生成: WHERE name = 'Alice'    ← Age=0 被忽略

// Map 条件（零值不会被忽略）
db.Where(map[string]interface{}{"name": "Alice", "age": 0}).Find(&users)
// 生成: WHERE name = 'Alice' AND age = 0

// --- 其他查询子句 ---

// Select 指定列
db.Select("name", "age").Find(&users)
db.Select("COALESCE(age, 0) as age").Find(&users)

// 排序
db.Order("age desc, name asc").Find(&users)

// 分页
db.Offset(10).Limit(20).Find(&users) // 跳过 10 条，取 20 条

// Distinct
db.Distinct("name").Find(&users)

// Count
var count int64
db.Model(&User{}).Where("age > ?", 18).Count(&count)

// Pluck：查询单列到切片
var names []string
db.Model(&User{}).Pluck("name", &names)

var ages []int
db.Model(&User{}).Pluck("age", &ages)

// --- Not / Or ---
db.Not("name = ?", "Alice").Find(&users)    // WHERE name <> 'Alice'
db.Where("age > ?", 18).Or("name = ?", "Admin").Find(&users)

// --- FirstOrInit / FirstOrCreate ---

// 找到就返回，找不到就用条件初始化（不写入 DB）
db.Where(User{Name: "new_user"}).Attrs(User{Age: 20}).FirstOrInit(&user)

// 找到就返回，找不到就创建（写入 DB）
db.Where(User{Name: "new_user"}).Attrs(User{Age: 20}).FirstOrCreate(&user)

// Assign：无论是否找到都赋值
db.Where(User{Name: "new_user"}).Assign(User{Age: 30}).FirstOrCreate(&user)
```

## 6. 高级查询

```go
// --- 子查询 ---
subQuery := db.Select("AVG(age)").Where("name LIKE ?", "A%").Table("users")
db.Where("age > (?)", subQuery).Find(&users)
// SELECT * FROM users WHERE age > (SELECT AVG(age) FROM users WHERE name LIKE 'A%')

// FROM 子查询
db.Table("(?) as u", db.Model(&User{}).Select("name", "age")).Where("age > ?", 18).Find(&users)

// --- Group / Having ---
type Result struct {
    Date  string
    Total int64
}
var results []Result
db.Model(&Order{}).
    Select("DATE(created_at) as date, SUM(amount) as total").
    Group("DATE(created_at)").
    Having("SUM(amount) > ?", 100).
    Scan(&results)

// --- Joins ---
type UserWithOrder struct {
    UserName   string
    OrderTotal float64
}
db.Model(&User{}).
    Select("users.name as user_name, SUM(orders.amount) as order_total").
    Joins("LEFT JOIN orders ON orders.user_id = users.id").
    Group("users.id").
    Scan(&results)

// --- Scan 到自定义结构体 ---
type SimpleUser struct {
    Name string
    Age  int
}
var simpleUsers []SimpleUser
db.Model(&User{}).Select("name", "age").Scan(&simpleUsers)

// --- FindInBatches：分批处理大量数据 ---
db.Where("active = ?", true).FindInBatches(&users, 100, func(tx *gorm.DB, batch int) error {
    for _, user := range users {
        // 处理每条记录
        _ = user
    }
    // 返回 error 会停止后续批次
    return nil
})

// --- Locking ---
// 悲观锁
db.Clauses(clause.Locking{Strength: "UPDATE"}).Find(&users)
// SELECT * FROM users FOR UPDATE

db.Clauses(clause.Locking{Strength: "SHARE"}).Find(&users)
// SELECT * FROM users FOR SHARE

// --- Optimizer / Index Hints ---
import "gorm.io/hints"

db.Clauses(hints.UseIndex("idx_name")).Find(&users)
db.Clauses(hints.ForceIndex("idx_name")).Where("name = ?", "Alice").Find(&users)

// --- Named Argument ---
db.Where("name = @name OR age = @age", sql.Named("name", "Alice"), sql.Named("age", 18)).Find(&users)

// --- Map 结果 ---
var resultMap []map[string]interface{}
db.Model(&User{}).Find(&resultMap)
```

## 7. 更新（Update）

```go
// --- 单字段更新 ---
db.Model(&user).Update("name", "new_name")
// UPDATE users SET name='new_name', updated_at=NOW() WHERE id=1

// 条件更新
db.Model(&User{}).Where("age < ?", 18).Update("active", false)

// --- 多字段更新（Struct）---
// 注意：struct 只更新非零值字段！
db.Model(&user).Updates(User{Name: "hello", Age: 0})
// 只更新 name，Age=0 被忽略

// --- 多字段更新（Map）---
// Map 可以更新零值
db.Model(&user).Updates(map[string]interface{}{"name": "hello", "age": 0})
// 更新 name 和 age

// --- Select / Omit 控制更新字段 ---
db.Model(&user).Select("Name", "Age").Updates(User{Name: "new", Age: 0})
// 强制更新 Name 和 Age（即使是零值）

db.Model(&user).Omit("Name").Updates(User{Name: "ignored", Age: 30})
// 跳过 Name，只更新 Age

// --- SQL 表达式 ---
db.Model(&product).Update("price", gorm.Expr("price * ? + ?", 2, 10))
// UPDATE products SET price = price * 2 + 10 WHERE id = 1

db.Model(&product).Updates(map[string]interface{}{
    "price": gorm.Expr("price + ?", 5),
    "stock": gorm.Expr("stock - ?", 1),
})

// --- 批量更新 ---
db.Model(&User{}).Where("active = ?", false).Updates(map[string]interface{}{"deleted": true})

// --- Save：全量保存所有字段（包括零值）---
user.Name = ""
user.Age = 0
db.Save(&user)
// UPDATE users SET name='', age=0, email='...', updated_at=NOW() WHERE id=1
// ⚠️ Save 会更新所有字段，慎用

// --- 不触发 Hook 和时间追踪的更新 ---
db.Model(&User{}).Where("id = ?", 1).UpdateColumn("name", "silent")
db.Model(&User{}).Where("id = ?", 1).UpdateColumns(map[string]interface{}{"name": "silent", "age": 30})
```

## 8. 删除（Delete）

```go
// --- 软删除（模型含 gorm.DeletedAt 字段时自动启用）---
db.Delete(&user)            // UPDATE users SET deleted_at=NOW() WHERE id=1
db.Delete(&user, 1)         // 指定主键
db.Delete(&User{}, []int{1, 2, 3}) // 批量删除

// 条件删除
db.Where("age < ?", 10).Delete(&User{})

// 软删除后的记录默认被排除
db.Find(&users) // 不含已软删除的

// 查找含软删除记录
db.Unscoped().Where("name = ?", "Alice").Find(&users)

// --- 永久删除（物理删除）---
db.Unscoped().Delete(&user)
// DELETE FROM users WHERE id = 1

// ⚠️ 不带条件的删除会被阻止（防止误删全表）
// db.Delete(&User{})  → 报错 ErrMissingWhereClause
// 如果确实需要删除全部：
db.Where("1 = 1").Delete(&User{})
// 或全局配置允许：gorm.Config{AllowGlobalUpdate: true}
```

## 9. 原生 SQL

```go
// --- Raw 查询 ---
var users []User
db.Raw("SELECT * FROM users WHERE age > ?", 18).Scan(&users)

var result struct {
    Total int64
    Avg   float64
}
db.Raw("SELECT COUNT(*) as total, AVG(age) as avg FROM users").Scan(&result)

// --- Exec 执行 ---
db.Exec("UPDATE users SET name = ? WHERE id = ?", "new_name", 1)
db.Exec("DROP TABLE IF EXISTS temp_users")

// --- SQL Builder（Table + Select + Where）---
var results []map[string]interface{}
db.Table("users").
    Select("name", "age").
    Where("id > ?", 5).
    Scan(&results)

// --- Row / Rows ---
row := db.Table("users").Where("name = ?", "Alice").Select("name", "age").Row()
var name string
var age int
row.Scan(&name, &age)

rows, _ := db.Model(&User{}).Where("age > ?", 18).Rows()
defer rows.Close()
for rows.Next() {
    var user User
    db.ScanRows(rows, &user) // 将 *sql.Rows 扫描到结构体
    fmt.Println(user)
}

// --- DryRun 模式：只生成 SQL 不执行 ---
stmt := db.Session(&gorm.Session{DryRun: true}).
    Where("name = ?", "Alice").
    Find(&User{}).Statement
fmt.Println(stmt.SQL.String())   // SQL 语句
fmt.Println(stmt.Vars)           // 绑定参数
```

## 10. 关联关系

```go
// ============ Belongs To ============
// 一个 Order 属于一个 User
type Order struct {
    gorm.Model
    Amount float64
    UserID uint    // 外键（默认命名：关联结构体名 + ID）
    User   User    // 关联字段
}

// 自定义外键
type Order2 struct {
    gorm.Model
    Amount    float64
    CreatorID uint
    Creator   User `gorm:"foreignKey:CreatorID"` // 指定外键
}

// ============ Has One ============
// 一个 User 拥有一个 Profile
type User struct {
    gorm.Model
    Name    string
    Profile Profile // has one
}
type Profile struct {
    gorm.Model
    UserID uint   // 外键
    Bio    string
    Avatar string
}

// ============ Has Many ============
// 一个 User 拥有多个 Order
type User struct {
    gorm.Model
    Name   string
    Orders []Order // has many
}

// ============ Many To Many ============
type User struct {
    gorm.Model
    Name      string
    Languages []Language `gorm:"many2many:user_languages;"` // 中间表
}
type Language struct {
    gorm.Model
    Name string
}

// 自定义中间表字段
type UserLanguage struct {
    UserID     uint `gorm:"primaryKey"`
    LanguageID uint `gorm:"primaryKey"`
    Skill      int  // 额外字段
}
// 需要用 SetupJoinTable 注册
db.SetupJoinTable(&User{}, "Languages", &UserLanguage{})

// ============ 多态关联 ============
type Comment struct {
    gorm.Model
    Content       string
    CommentableID   uint
    CommentableType string
}
type Article struct {
    gorm.Model
    Title    string
    Comments []Comment `gorm:"polymorphic:Commentable;"` // 多态
}
type Video struct {
    gorm.Model
    URL      string
    Comments []Comment `gorm:"polymorphic:Commentable;"`
}

// ============ Association Mode（关联操作）============

// 创建时自动写关联
db.Create(&User{
    Name:   "Alice",
    Orders: []Order{{Amount: 100}, {Amount: 200}},
})

// 添加关联
db.Model(&user).Association("Orders").Append(&Order{Amount: 300})

// 替换关联
db.Model(&user).Association("Languages").Replace(&Language{Name: "Go"}, &Language{Name: "Rust"})

// 删除关联（只解除关系，不删记录）
db.Model(&user).Association("Orders").Delete(&order)

// 清空关联
db.Model(&user).Association("Orders").Clear()

// 关联计数
count := db.Model(&user).Association("Orders").Count()

// 查找关联
var orders []Order
db.Model(&user).Association("Orders").Find(&orders)
```

## 11. 预加载

```go
// --- Preload（每个关联单独一条 SQL）---
var users []User

// 基本预加载
db.Preload("Orders").Find(&users)
// SELECT * FROM users;
// SELECT * FROM orders WHERE user_id IN (1,2,3);

// 条件预加载
db.Preload("Orders", "amount > ?", 100).Find(&users)

// 自定义预加载
db.Preload("Orders", func(db *gorm.DB) *gorm.DB {
    return db.Where("amount > ?", 50).Order("created_at DESC").Limit(5)
}).Find(&users)

// 嵌套预加载
db.Preload("Orders.Items").Find(&users)             // Orders → Items
db.Preload("Orders").Preload("Profile").Find(&users) // 同时预加载多个

// 全部预加载
db.Preload(clause.Associations).Find(&users)

// --- Joins 预加载（单条 SQL，适合一对一/Belongs To）---
db.Joins("Profile").Find(&users)
// SELECT users.*, profiles.* FROM users LEFT JOIN profiles ON ...

// 条件 Joins 预加载
db.Joins("Profile", db.Where(&Profile{Bio: "engineer"})).Find(&users)

// ⚠️ Preload vs Joins:
// Preload: 多条 SQL，适合一对多、多对多
// Joins:   单条 SQL JOIN，适合一对一、belongs to，数据量大时更高效
```

## 12. 事务

```go
// --- 自动事务（推荐）---
err := db.Transaction(func(tx *gorm.DB) error {
    if err := tx.Create(&User{Name: "Alice"}).Error; err != nil {
        return err // 返回 error 自动回滚
    }
    if err := tx.Create(&Order{Amount: 100, UserID: 1}).Error; err != nil {
        return err
    }
    return nil // 返回 nil 自动提交
})

// --- 嵌套事务 / SavePoint ---
db.Transaction(func(tx *gorm.DB) error {
    tx.Create(&user1)

    tx.Transaction(func(tx2 *gorm.DB) error {
        tx2.Create(&user2)
        return errors.New("rollback inner") // user2 回滚
    })

    // user1 仍然会提交
    return nil
})

// --- SavePoint / RollbackTo ---
db.Transaction(func(tx *gorm.DB) error {
    tx.Create(&user1)

    tx.SavePoint("sp1")
    tx.Create(&user2)
    tx.RollbackTo("sp1") // user2 回滚

    return nil // user1 提交
})

// --- 手动事务 ---
tx := db.Begin()
if err := tx.Error; err != nil {
    return err
}

if err := tx.Create(&user).Error; err != nil {
    tx.Rollback()
    return err
}

if err := tx.Create(&order).Error; err != nil {
    tx.Rollback()
    return err
}

tx.Commit()

// ⚠️ 手动事务忘记 Rollback 或 Commit 会导致连接泄漏
// 推荐用 defer：
tx := db.Begin()
defer func() {
    if r := recover(); r != nil {
        tx.Rollback()
    }
}()
```

## 13. Hook（钩子）

```go
// Hook 在特定操作前后自动执行
// 创建顺序：BeforeSave → BeforeCreate → 写入DB → AfterCreate → AfterSave
// 更新顺序：BeforeSave → BeforeUpdate → 写入DB → AfterUpdate → AfterSave
// 删除顺序：BeforeDelete → 删除DB → AfterDelete
// 查询顺序：AfterFind

// 返回 error 会中断操作并回滚（事务中）

type User struct {
    gorm.Model
    Name string
    UUID string
    Age  int
}

// 创建前自动生成 UUID
func (u *User) BeforeCreate(tx *gorm.DB) error {
    u.UUID = uuid.New().String()
    return nil
}

// 创建前校验
func (u *User) BeforeSave(tx *gorm.DB) error {
    if u.Age < 0 {
        return errors.New("age cannot be negative")
    }
    return nil
}

// 查询后处理
func (u *User) AfterFind(tx *gorm.DB) error {
    // 如：解密敏感字段、计算派生字段
    return nil
}

// 删除前检查
func (u *User) BeforeDelete(tx *gorm.DB) error {
    // 如：检查是否有关联数据
    var count int64
    tx.Model(&Order{}).Where("user_id = ?", u.ID).Count(&count)
    if count > 0 {
        return errors.New("cannot delete user with orders")
    }
    return nil
}

// 跳过 Hook：使用 Session
db.Session(&gorm.Session{SkipHooks: true}).Create(&user)
```

## 14. Scopes 与链式调用

```go
// Scope 是可复用的查询片段，类型为 func(*gorm.DB) *gorm.DB

// 分页
func Paginate(page, pageSize int) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        if page <= 0 {
            page = 1
        }
        if pageSize <= 0 {
            pageSize = 10
        }
        return db.Offset((page - 1) * pageSize).Limit(pageSize)
    }
}

// 状态过滤
func Active(db *gorm.DB) *gorm.DB {
    return db.Where("active = ?", true)
}

func CreatedAfter(t time.Time) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        return db.Where("created_at > ?", t)
    }
}

// 按角色过滤
func WithRole(role string) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        return db.Where("role = ?", role)
    }
}

// 组合使用
db.Scopes(Active, WithRole("admin"), Paginate(1, 20)).Find(&users)

db.Scopes(
    CreatedAfter(time.Now().AddDate(0, -1, 0)),
    Paginate(2, 50),
).Find(&orders)

// 条件性 Scope
func FilterByName(name string) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        if name != "" {
            return db.Where("name LIKE ?", "%"+name+"%")
        }
        return db
    }
}
```

## 15. 泛型 API（v1.30+）

```go
// GORM v1.30 引入泛型 API：类型安全、强制 context、返回值直接是目标类型
import (
    "context"
    "gorm.io/gorm"
)

type Product struct {
    gorm.Model
    Code  string
    Price uint
}

func main() {
    db, _ := gorm.Open(sqlite.Open("test.db"), &gorm.Config{})
    ctx := context.Background()

    // Create
    err := gorm.G[Product](db).Create(ctx, &Product{Code: "D42", Price: 100})

    // 查询单条 — 返回 (Product, error)，不需要提前声明变量
    product, err := gorm.G[Product](db).Where("code = ?", "D42").First(ctx)

    // 查询多条 — 返回 ([]Product, error)
    products, err := gorm.G[Product](db).Where("price > ?", 50).Find(ctx)

    // 更新
    err = gorm.G[Product](db).Where("id = ?", product.ID).Update(ctx, "Price", 200)

    // 多字段更新
    err = gorm.G[Product](db).Where("id = ?", product.ID).Updates(ctx, Product{Code: "E42", Price: 300})

    // 删除
    err = gorm.G[Product](db).Where("id = ?", product.ID).Delete(ctx)
}

// 对比传统 API：
// 传统：db.First(&product, 1)       → 通过指针回填，无编译期类型检查
// 泛型：product, err := gorm.G[Product](db).First(ctx)  → 返回值即类型，编译期安全
```

## 16. Session 与上下文

```go
// --- Session：创建新的 DB 会话，避免条件污染 ---

// NewDB 模式：不继承之前的条件
tx := db.Where("name = ?", "Alice").Session(&gorm.Session{NewDB: true})
```

## 17. 自定义数据类型

```go
import (
    "database/sql/driver"
    "encoding/json"
)

// --- 实现 Scanner 和 Valuer 接口即可自定义类型 ---

// JSON 类型
type JSON map[string]interface{}

// 从数据库读取时调用
func (j *JSON) Scan(value interface{}) error {
    bytes, ok := value.([]byte)
    if !ok {
        return errors.New("type assertion to []byte failed")
    }
    return json.Unmarshal(bytes, j)
}

// 写入数据库时调用
func (j JSON) Value() (driver.Value, error) {
    return json.Marshal(j)
}

// 使用
type Product struct {
    gorm.Model
    Name string
    Attrs JSON `gorm:"type:json"` // MySQL JSON 列
}

db.Create(&Product{
    Name:  "Phone",
    Attrs: JSON{"color": "black", "storage": "128GB"},
})

// --- StringSlice：存储为逗号分隔字符串 ---
type StringSlice []string

func (s *StringSlice) Scan(value interface{}) error {
    str, ok := value.(string)
    if !ok {
        bytes, ok := value.([]byte)
        if !ok {
            return errors.New("unsupported type")
        }
        str = string(bytes)
    }
    *s = strings.Split(str, ",")
    return nil
}

func (s StringSlice) Value() (driver.Value, error) {
    return strings.Join(s, ","), nil
}

// --- GORM 内置 Serializer（更便捷）---
type UserPreference struct {
    gorm.Model
    Settings JSON `gorm:"serializer:json"` // 自动 JSON 序列化/反序列化
}
```

## 18. 性能优化

```go
// --- 1. 跳过默认事务（只读场景约提升 30%）---
db, _ := gorm.Open(mysql.Open(dsn), &gorm.Config{
    SkipDefaultTransaction: true,
})

// --- 2. 预编译缓存 ---
db, _ = gorm.Open(mysql.Open(dsn), &gorm.Config{
    PrepareStmt: true, // 缓存预编译语句，减少 DB 解析开销
})

// --- 3. 只查需要的列 ---
db.Select("id", "name").Find(&users) // 避免 SELECT *

// --- 4. 批量插入 ---
db.CreateInBatches(&users, 100) // 每批 100 条

// --- 5. FindInBatches 处理大量数据 ---
db.Where("active = ?", true).FindInBatches(&users, 500, func(tx *gorm.DB, batch int) error {
    // 每批 500 条
    return nil
})

// --- 6. 用 Map 查询避免反射开销 ---
var results []map[string]interface{}
db.Model(&User{}).Find(&results)

// --- 7. 索引优化 ---
type User struct {
    gorm.Model
    Email  string `gorm:"uniqueIndex"`                    // 唯一索引
    Name   string `gorm:"index"`                          // 普通索引
    Age    int    `gorm:"index:idx_age_active"`            // 复合索引
    Active bool   `gorm:"index:idx_age_active"`
}

// --- 8. 使用 Joins 替代 Preload（一对一场景）---
// Preload 发 2 条 SQL，Joins 只发 1 条
db.Joins("Profile").Find(&users)

// --- 9. 连接池调优 ---
sqlDB, _ := db.DB()
sqlDB.SetMaxIdleConns(10)
sqlDB.SetMaxOpenConns(100)
sqlDB.SetConnMaxLifetime(time.Hour)
```

## 20. 常见易错点

```go
// ❌ 1. struct 更新零值被忽略
db.Model(&user).Updates(User{Name: "Alice", Age: 0})
// Age=0 不会更新！
// ✅ 用 Map：
db.Model(&user).Updates(map[string]interface{}{"name": "Alice", "age": 0})
// ✅ 或 Select 强制指定：
db.Model(&user).Select("Name", "Age").Updates(User{Name: "Alice", Age: 0})

// ❌ 2. 链式调用条件污染
// db 是全局的，Where 条件会累积！
db.Where("name = ?", "Alice") // 条件会残留
db.Find(&users)               // 带上了 name='Alice'
// ✅ 使用 Session 或每次创建新的 tx
tx := db.Session(&gorm.Session{NewDB: true})

// ❌ 3. First 找不到记录返回 ErrRecordNotFound
err := db.First(&user, 999).Error
// err == gorm.ErrRecordNotFound
// ✅ 用 Find 不会报错（返回空切片），或显式判断：
if errors.Is(err, gorm.ErrRecordNotFound) {
    // 处理未找到
}

// ❌ 4. 忘记处理 error
db.Create(&user) // 如果失败了你永远不知道
// ✅
if err := db.Create(&user).Error; err != nil {
    log.Fatal(err)
}

// ❌ 5. Save 更新全部字段（包括零值）
user.Age = 0
db.Save(&user) // Age 被更新为 0，Name 等也重写
// ✅ 用 Updates 只更新需要的字段

// ❌ 6. 无条件删除/更新被阻止
db.Delete(&User{})                          // 报错：ErrMissingWhereClause
db.Model(&User{}).Update("active", false)   // 报错
// ✅
db.Where("1 = 1").Delete(&User{})

// ❌ 7. Preload N+1 问题
for _, user := range users {
    db.Model(&user).Association("Orders").Find(&orders) // 每个用户一条 SQL
}
// ✅ 一次预加载
db.Preload("Orders").Find(&users)

// ❌ 8. 软删除后忘记 Unscoped
db.Find(&users) // 不含已删除记录
// ✅ 需要查已删除的：
db.Unscoped().Find(&users)

// ❌ 9. AutoMigrate 期望它能删列
// AutoMigrate 只加不减，不会删除多余列或降级列类型
// ✅ 生产环境用专用迁移工具（golang-migrate / goose 等）

// ❌ 10. 并发使用同一个 *gorm.DB 实例的 Where
// *gorm.DB 链式调用是值语义（新 Statement），但要注意不能在 goroutine 间共享同一个 tx
// ✅ 每个 goroutine 使用 db.Session(&gorm.Session{}) 或直接用全局 db（全局 db 是安全的）
```
